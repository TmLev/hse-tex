\begin{itemize}[<+->]
    \item ``Меньше или равно'' и ``больше или равно''
    \item Косые чаще выглядят лучше
    \item Заменим по умолчанию
\end{itemize}

\begin{onlyenv}<1>
    \begin{align*}
        x \leq y \quad y \geq z
    \end{align*}

    \begin{block}{Код}
        \begin{lstlisting}
    \begin{align*}
        x \leq y \quad y \geq z
    \end{align*}
        \end{lstlisting}
    \end{block}
\end{onlyenv}

\begin{onlyenv}<2>
    \begin{align*}
        x \leqslant y \quad y \geqslant z
    \end{align*}
    
    \begin{block}{Код}
        \begin{lstlisting}
    \begin{align*}
        x \leqslant y \quad y \geqslant z
    \end{align*}
        \end{lstlisting}
    \end{block}
\end{onlyenv}

\begin{onlyenv}<3>
    {
        \renewcommand{\leq}{\leqslant}
        \renewcommand{\geq}{\geqslant}
        
        \begin{align*}
            x \leq y \quad y \geq z
        \end{align*}
    }
    
    \begin{block}{Код}
        \begin{lstlisting}
    \renewcommand{\leq}{\leqsant}
    \renewcommand{\geq}{\geqslant}
    
    \begin{align*}
        x \leq y \quad y \geq z
    \end{align*}
        \end{lstlisting}
    \end{block}
\end{onlyenv}

\end{frame}
